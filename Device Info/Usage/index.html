    Usage

    Usage example of how to use the module.

    For API Reference, see the API Reference page.

    Log device info and changes to the console¶

    local module = require(game.ReplicatedStorage.DeviceInfo)

local platform, deviceType, screenResolution, inputType, screenOrientation, graphicsQuality

local mPlatform, mDeviceType, mScreenResolution = module.getDevicePlatform(), module.getDeviceType(), module.getDeviceScreenSize()
local mInputType = module.getInputType()
local mScreenOrientation = module.getDeviceOrientation()
local mGraphicsQuality = module.getGraphicsQuality()

if mPlatform == module.Enum.PlatformType.Computer then
    platform = "Computer"
elseif mPlatform == module.Enum.PlatformType.Console then
    platform = "Console"
elseif mPlatform == module.Enum.PlatformType.Mobile then
    platform = "Mobile"
end

if mDeviceType == module.Enum.DeviceType.Computer then
    deviceType = "Computer"
elseif mDeviceType == module.Enum.DeviceType.Console then
    deviceType = "Console"
elseif mDeviceType == module.Enum.DeviceType.Phone then
    deviceType = "Phone"
elseif mDeviceType == module.Enum.DeviceType.Tablet then
    deviceType =  "Tablet"
elseif mDeviceType == module.Enum.DeviceType.TouchscreenComputer then
    deviceType =  "Touchscreen computer"
end

if mInputType == module.Enum.InputType.Gamepad then
    inputType = "Gamepad"
elseif mInputType == module.Enum.InputType.Keyboard then
    inputType = "Keyboard"
elseif mInputType == module.Enum.InputType.Mouse then
    inputType = "Mouse"
elseif mInputType == module.Enum.InputType.KeyboardMouse then
    inputType =  "Keyboard and mouse"
elseif mInputType == module.Enum.InputType.Touchscreen then
    inputType =  "Touchscreen"
elseif mInputType == module.Enum.InputType.VR then
    inputType =  "VR"
end

if mPlatform == module.Enum.PlatformType.Mobile then
    local mScreenOrientation = module.getDeviceOrientation()
    if mScreenOrientation == module.Enum.DeviceOrientation.Landscape then
        screenOrientation = "Lanadscape"
    elseif mScreenOrientation == module.Enum.DeviceOrientation.Portrait then
        screenOrientation = "Portrait"
    end
else
    screenOrientation = "Only supported on Mobile devices"
end

if mGraphicsQuality == module.Enum.GraphicsQuality.Automatic then
    graphicsQuality = "Automatic"
elseif mGraphicsQuality == module.Enum.GraphicsQuality.Level01 then
    graphicsQuality = "Level 01"
elseif mGraphicsQuality == module.Enum.GraphicsQuality.Level02 then
    graphicsQuality = "Level 02"
elseif mGraphicsQuality == module.Enum.GraphicsQuality.Level03 then
    graphicsQuality = "Level 03"
elseif mGraphicsQuality == module.Enum.GraphicsQuality.Level04 then
    graphicsQuality = "Level 04"
elseif mGraphicsQuality == module.Enum.GraphicsQuality.Level05 then
    graphicsQuality = "Level 05"
elseif mGraphicsQuality == module.Enum.GraphicsQuality.Level06 then
    graphicsQuality = "Level 06"
elseif mGraphicsQuality == module.Enum.GraphicsQuality.Level07 then
    graphicsQuality = "Level 07"
elseif mGraphicsQuality == module.Enum.GraphicsQuality.Level08 then
    graphicsQuality = "Level 08"
elseif mGraphicsQuality == module.Enum.GraphicsQuality.Level09 then
    graphicsQuality = "Level 09"
elseif mGraphicsQuality == module.Enum.GraphicsQuality.Level10 then
    graphicsQuality = "Level 10"
end

print("----------------------------------")
print(" Platform: "..platform)
print(" Device type: "..deviceType)
print(" Screen resolution: "..mScreenResolution.X.." x "..mScreenResolution.Y)
print(" Input type: "..inputType)
print(" Graphics quality level: "..graphicsQuality)
print(" Screen orientation: "..screenOrientation)
print("----------------------------------")

module.inputTypeChanged:Connect(function(newInputType)
    if newInputType == module.Enum.InputType.Gamepad then
        print("Input type changed to Gamepad")
    elseif newInputType == module.Enum.InputType.Keyboard then
        print("Input type changed to Keyboard")
    elseif newInputType == module.Enum.InputType.Mouse then
        print("Input type changed to Mouse")
    elseif newInputType == module.Enum.InputType.KeyboardMouse then
        print("Input type changed to Keyboard and mouse")
    elseif newInputType == module.Enum.InputType.Touchscreen then
        print("Input type changed to Touchscreen")
    elseif newInputType == module.Enum.InputType.VR then
        print("Input type changed to VR")
    end
end)
print("Listening for input type changes.")

module.screenSizeChanged:Connect(function(newSize)
    print("Resolution changed: "..newSize.X.." x "..newSize.Y)
end)
print("Listening for resolution changes.")

module.screenOrientationChanged:Connect(function(newOrientation)
    if newOrientation == module.Enum.DeviceOrientation.Portrait then
        print("Screen orientation changed to Portrait")
    elseif newOrientation == module.Enum.DeviceOrientation.Landscape then
        print("Screen orientation changed to Landscape")
    end
end)
print("Listening for orientation changes.")

module.graphicsQualityChanged:Connect(function(newQuality)
    if newQuality == module.Enum.GraphicsQuality.Automatic then
        print("Graphic quality level changed to Automatic")
    elseif newQuality == module.Enum.GraphicsQuality.Level01 then
        print("Graphic quality level changed to Level 01")
    elseif newQuality == module.Enum.GraphicsQuality.Level02 then
        print("Graphic quality level changed to Level 02")
    elseif newQuality == module.Enum.GraphicsQuality.Level03 then
        print("Graphic quality level changed to Level 03")
    elseif newQuality == module.Enum.GraphicsQuality.Level04 then
        print("Graphic quality level changed to Level 04")
    elseif newQuality == module.Enum.GraphicsQuality.Level05 then
        print("Graphic quality level changed to Level 05")
    elseif newQuality == module.Enum.GraphicsQuality.Level06 then
        print("Graphic quality level changed to Level 06")
    elseif newQuality == module.Enum.GraphicsQuality.Level07 then
        print("Graphic quality level changed to Level 07")
    elseif newQuality == module.Enum.GraphicsQuality.Level08 then
        print("Graphic quality level changed to Level 08")
    elseif newQuality == module.Enum.GraphicsQuality.Level09 then
        print("Graphic quality level changed to Level 09")
    elseif newQuality == module.Enum.GraphicsQuality.Level10 then
        print("Graphic quality level changed to Level 10")
    end
end)
print("Listening for graphic quality changes.")
